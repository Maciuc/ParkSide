CREATE TABLE Players(
	Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	First_Name NVARCHAR(200) NOT NULL,
	Last_Name NVARCHAR(200) NOT NULL,
	Height NVARCHAR(10),
	Number NVARCHAR(10),
	BirthDate DATETIME,
	Nationality NVARCHAR(100),
	Description NVARCHAR(500),
	ImageURL NVARCHAR(500),
	IsDeleted BIT NOT NULL DEFAULT(0)
);

CREATE TABLE Stuff(
	Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	First_Name NVARCHAR(200) NOT NULL,
	Last_Name NVARCHAR(200) NOT NULL,
	Height NVARCHAR(200),
	BirthDate DATETIME,
	Nationality NVARCHAR(100),
	ImageURL NVARCHAR(500),
	Description NVARCHAR(500),
	IsDeleted BIT NOT NULL DEFAULT(0)
);

CREATE TABLE Sponsors(
	Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	Name NVARCHAR(200) NOT NULL,
	Link NVARCHAR(MAX),
	ImageURL NVARCHAR(500),
	IsDeleted BIT NOT NULL DEFAULT(0)
);

CREATE TABLE News(
	Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	Name NVARCHAR(200) NOT NULL,
	Description NVARCHAR(500),
	Content NVARCHAR(MAX),
	ImageURL NVARCHAR(500),
	IsPublished BIT NOT NULL DEFAULT(0),
	IsPrimary BIT NOT NULL DEFAULT(0)
	PublishedDate DATETIME,
	IsDeleted BIT NOT NULL DEFAULT(0)
);

CREATE TABLE SocialMedias(
	Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	Name NVARCHAR(200) NOT NULL,
	Link NVARCHAR(500),
	Platform NVARCHAR(200),
	IsDeleted BIT NOT NULL DEFAULT(0)
);

CREATE TABLE Championships(
	Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	Name NVARCHAR(200) NOT NULL,
	ImageURL NVARCHAR(500),
	IsDeleted BIT NOT NULL DEFAULT(0)
);

CREATE TABLE Teams(
	Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	Name NVARCHAR(200) NOT NULL,
	ImageURL NVARCHAR(500),
	IsDeleted BIT NOT NULL DEFAULT(0)
);

CREATE TABLE Matches(
	Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	ChampionshipId INT REFERENCES Championships(Id) NOT NULL,
	EnemyTeamId INT REFERENCES Teams(Id) NOT NULL,
	Location NVARCHAR(500),
	MatchDate DATETIME,
	MatchHour NVARCHAR(20),
	PlayingHome BIT NOT NULL DEFAULT(0),
	IsFinished BIT NOT NULL DEFAULT(0),
	EnemyTeamPoints NVARCHAR(10),
	MainTeamPoints NVARCHAR(10),
	IsDeleted BIT NOT NULL DEFAULT(0)
);

CREATE TABLE Trofees(
	Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	Name NVARCHAR(200) NOT NULL,
	ImageURL NVARCHAR(500),
	IsDeleted BIT NOT NULL DEFAULT(0)
);

CREATE TABLE PlayersHistory(
	Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	PlayerId INT REFERENCES Players(Id) NOT NULL,
	ChampionshipId INT REFERENCES Championships(Id) NOT NULL,
	Year NVARCHAR(5),
	PlayerRole NVARCHAR(50),
	TeamName NVARCHAR(200),
	IsDeleted BIT NOT NULL DEFAULT(0)
);

CREATE TABLE PlayersTrofees(
	Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	PlayerHistoryId INT REFERENCES PlayersHistory(Id) NOT NULL,
	TrofeeId INT REFERENCES Trofees(Id) NOT NULL,
	IsDeleted BIT NOT NULL DEFAULT(0)
);

CREATE TABLE StuffHistory(
	Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY,
	StuffId INT REFERENCES Stuff(Id) NOT NULL,
	ChampionshipId INT REFERENCES Championships(Id) NOT NULL,
	Year NVARCHAR(5),
	TeamName NVARCHAR(200),
	IsDeleted BIT NOT NULL DEFAULT(0)
);
